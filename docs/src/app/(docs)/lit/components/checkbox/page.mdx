## Anatomy

```ts title="Anatomy"
import { html, render } from 'lit';
import { Checkbox } from '@base-ui/lit/checkbox';

render(
  Checkbox.Root({
    children: Checkbox.Indicator({ children: html`<span>✓</span>` }),
  }),
  mountNode,
);
```
