## Anatomy

```ts title="Anatomy"
import { render } from 'lit';
import { Field } from '@base-ui/lit/field';
import { Form } from '@base-ui/lit/form';

render(
  Form({
    children: Field.Root({
      children: [
        Field.Label({ children: 'Homepage' }),
        Field.Control({ required: true }),
        Field.Error(),
      ],
    }),
  }),
  mountNode,
);
```
