## Anatomy

```ts title="Anatomy"
import { html, render } from 'lit';
import { Checkbox } from '@base-ui/lit/checkbox';
import { CheckboxGroup } from '@base-ui/lit/checkbox-group';

render(
  CheckboxGroup({
    children: html` ${Checkbox.Root({ value: 'apple' })} ${Checkbox.Root({ value: 'orange' })} `,
  }),
  mountNode,
);
```
