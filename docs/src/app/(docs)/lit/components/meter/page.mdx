## Anatomy

```ts title="Anatomy"
import { render } from 'lit';
import { Meter } from '@base-ui/lit/meter';

render(
  Meter.Root({
    value: 72,
    children: [Meter.Track({ children: Meter.Indicator() }), Meter.Value(), Meter.Label()],
  }),
  mountNode,
);
```
