## Anatomy

```ts title="Anatomy"
import { render } from 'lit';
import { PreviewCard } from '@base-ui/lit/preview-card';

render(
  PreviewCard.Root({
    children: [
      PreviewCard.Trigger({ href: '#', children: 'Typography' }),
      PreviewCard.Portal({
        children: PreviewCard.Positioner({
          children: PreviewCard.Popup({
            children: PreviewCard.Arrow(),
          }),
        }),
      }),
    ],
  }),
  mountNode,
);
```
