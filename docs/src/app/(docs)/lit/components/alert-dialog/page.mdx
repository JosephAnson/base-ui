## Anatomy

```ts title="Anatomy"
import { render } from 'lit';
import { AlertDialog } from '@base-ui/lit/alert-dialog';

render(
  AlertDialog.Root({
    children: [
      AlertDialog.Trigger({ children: 'Discard draft' }),
      AlertDialog.Portal({
        children: [
          AlertDialog.Backdrop(),
          AlertDialog.Popup({
            children: [
              AlertDialog.Title({ children: 'Discard draft?' }),
              AlertDialog.Description({ children: "You can't undo this action." }),
              AlertDialog.Close({ children: 'Cancel' }),
            ],
          }),
        ],
      }),
    ],
  }),
  mountNode,
);
```
