## Anatomy

```ts title="Anatomy"
import { render } from 'lit';
import { Field } from '@base-ui/lit/field';

render(
  Field.Root({
    children: [
      Field.Label({ children: 'Name' }),
      Field.Control({ required: true }),
      Field.Error({ match: 'valueMissing', children: 'Required' }),
    ],
  }),
  mountNode,
);
```
