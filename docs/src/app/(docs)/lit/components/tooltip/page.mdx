## Anatomy

```ts title="Anatomy"
import { render } from 'lit';
import { Tooltip } from '@base-ui/lit/tooltip';

render(
  Tooltip.Provider({
    children: Tooltip.Root({
      children: [
        Tooltip.Trigger({ children: 'Open' }),
        Tooltip.Portal({
          children: Tooltip.Positioner({
            children: Tooltip.Popup({
              children: Tooltip.Arrow(),
            }),
          }),
        }),
      ],
    }),
  }),
  mountNode,
);
```
