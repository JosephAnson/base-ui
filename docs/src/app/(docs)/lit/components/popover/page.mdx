## Anatomy

```ts title="Anatomy"
import { render } from 'lit';
import { Popover } from '@base-ui/lit/popover';

render(
  Popover.Root({
    children: [
      Popover.Trigger({ children: 'Open' }),
      Popover.Portal({
        children: Popover.Positioner({
          children: Popover.Popup({ children: 'Content' }),
        }),
      }),
    ],
  }),
  mountNode,
);
```
