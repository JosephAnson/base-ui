## Anatomy

```ts title="Anatomy"
import { render } from 'lit';
import { Accordion } from '@base-ui/lit/accordion';

render(
  Accordion.Root({
    children: Accordion.Item({
      value: 'item-1',
      children: [
        Accordion.Header({
          children: Accordion.Trigger({ children: 'Section title' }),
        }),
        Accordion.Panel({ children: 'Section content' }),
      ],
    }),
  }),
  mountNode,
);
```
