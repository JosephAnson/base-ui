## Anatomy

```ts title="Anatomy"
import { render } from 'lit';
import { Collapsible } from '@base-ui/lit/collapsible';

render(
  Collapsible.Root({
    children: [
      Collapsible.Trigger({ children: 'Toggle section' }),
      Collapsible.Panel({ children: 'Section content' }),
    ],
  }),
  mountNode,
);
```
