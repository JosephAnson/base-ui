## Anatomy

```ts title="Anatomy"
import { render } from 'lit';
import { Progress } from '@base-ui/lit/progress';

render(
  Progress.Root({
    value: 60,
    children: [Progress.Track({ children: Progress.Indicator() }), Progress.Value()],
  }),
  mountNode,
);
```
