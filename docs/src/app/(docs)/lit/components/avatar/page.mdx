## Anatomy

```ts title="Anatomy"
import { html, render } from 'lit';
import { Avatar } from '@base-ui/lit/avatar';

render(
  Avatar.Root({
    children: html`
      ${Avatar.Image({ src: '/avatar.jpg', alt: 'Ada Lovelace' })}
      ${Avatar.Fallback({ children: 'AL' })}
    `,
  }),
  mountNode,
);
```
